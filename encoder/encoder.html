encoder()

コード

ヘッダファイル

    
#include "mbed.h"

class ENC {
    InterruptIn A_in;
    InterruptIn B_in;
    DigitalIn A_dig;
    DigitalIn B_dig;
    void A_rise();
    void B_rise();
    void A_fall();
    void B_fall();
    public:
        ENC(PinName inA, PinName inB, PinName A, PinName B);
        int enc_count = 0;
        void rise_fall();    
};

実装

    
#include "enc.h"

ENC::ENC(PinName inA, PinName inB, PinName A, PinName B)
    : A_in(inA), B_in(inB), A_dig(A), B_dig(B)  // メンバを初期化子リストで初期化！
{
    // 初期化処理があればここに書く
}

void ENC::A_rise()
{
    if(A_dig == 1){
        enc_count--;
    }else{
        enc_count++;
    }
}

void ENC::B_rise()
{
    if(B_dig == 1){
        enc_count++;
    }else{
        enc_count--;
    }
}

void ENC::A_fall()
{
    if(A_dig == 1){
        enc_count++;
    }else{
        enc_count--;
    }
}

void ENC::B_fall()
{
    if(B_dig == 1){
        enc_count--;
    }else{
        enc_count++;
    }
}

void ENC::rise_fall() {
    A_in.rise(callback(this, &ENC::A_rise));
    A_in.fall(callback(this, &ENC::A_fall));
    B_in.rise(callback(this, &ENC::B_rise));
    B_in.fall(callback(this, &ENC::B_fall));
}
  • それぞれやっていることはほぼ同じでピンのHIGHとLOWを見てカウンタを増やしているよ
